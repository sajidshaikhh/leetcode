containsDuplicate

class Solution(object):
    def containsDuplicate(self,nums):
        nums.sort()
        ist = False
        if len(nums) == 1:
            return False
        for i in range(len(nums)):
            if nums[i-1] == nums[i]:
                ist = True
        return ist

Tow Sum

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        n = len(nums)
        for i in range(n - 1):
            for j in range(i + 1, n):
                if nums[i] + nums[j] == target:
                    return [i, j]
        return []
